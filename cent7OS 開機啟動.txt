sudo vi /etc/systemd/system/{name}.service

生成文本:
[Unit]
Description=web.Core Server

[Service]
WorkingDirectory=/vagrant
ExecStart=/usr/bin/dotnet /vagrant/langPJ/languagePJ.dll
Restart=always
# Restart service after 10 seconds if the dotnet service crashes:
RestartSec=10
SyslogIdentifier=web-core
User=vagrant
Environment=ASPNETCORE_ENVIRONMENT=Production
Environment=DOTNET_PRINT_TELEMETRY_MESSAGE=false

[Install]
WantedBy=multi-user.target

//----------------

sudo systemctl enable kestrel-web.service //註冊 開機啟動
sudo systemctl disable kestrel-web.service //remove 註冊
sudo systemctl stop kestrel-web.service 
sudo systemctl start kestrel-web.service//立即啟用
sudo systemctl status kestrel-web.service
